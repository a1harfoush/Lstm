        "def lstm_step(x, h_prev, c_prev):\n",
        "\n",
        "\n",
        "    f = sigmoid(Wf * x + Uf * h_prev + bf)\n",
        "\n",
        "\n",
        "    i = sigmoid(Wi * x + Ui * h_prev + bi)\n",
        "\n",
        "\n",
        "    c_hat = tanh(Wc * x + Uc * h_prev + bc)\n",
        "\n",
        "\n",
        "    c = f * c_prev + i * c_hat\n",
        "\n",
        "\n",
        "    o = sigmoid(Wo * x + Uo * h_prev + bo)\n",
        "\n",
        "\n",
        "    h = o * tanh(c)\n",
        "\n",
        "    return f, i, c_hat, c, o, h"
      ]
